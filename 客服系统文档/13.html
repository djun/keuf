如何自定义发送商品消息

在电商场景下很多的需求是在用户点击咨询的时候，将当前的咨询的商品一并发送给客服，方便客服快速定位商品，针对解决问题。

如何在 AiService中增加这样的功能？

在访客正确的分配到访客的地方 AiService.cli.v2.js 的 userInitHandle 正确返回，看到 0 == data.code ，此时我们在处理的内容中自己构建当前上面的信息

socket.send(JSON.stringify({
    cmd: "chatMessage",
    data: {
        from_name: customer.name,
        from_avatar: customer.avatar,
        from_id: customer.uid,
        to_id: kefuCode,
        to_name: kefuName,
        content: 'img[https://g-search1.alicdn.com/img/bao/uploaded/i4/imgextra/i4/45646959/O1CN01vxwBYY21HGpyruFZe_!!0-saturn_solar.jpg_250x250.jpg_.webp][p]当天发【24期分期/送无线充】Apple/苹果 iPhone 11 手机现货4G全网通新款iphone11 pro max国行正品保证xs[/p][p]￥3,999[/p]',
        seller_code: seller
    }
}));

可以看到，我这里搞了3个标签

img 包裹的 商品图片

img[https://g-search1.alicdn.com/img/bao/uploaded/i4/imgextra/i4/45646959/O1CN01vxwBYY21HGpyruFZe_!!0-saturn_solar.jpg_250x250.jpg_.webp]

第一个p标签是 商品的简介

第二个p标签是 商品的价格

[p]￥3,999[/p]

replaceContent = function(content) {
    // 支持的html标签
    var html = function (end) {
        return new RegExp('\\n*\\[' + (end || '') + '(pre|div|span|img|br|a|em|font|strong|p|table|thead|th|tbody|tr|td|ul|li|ol|li|dl|dt|dd|h2|h3|h4|h5)([\\s\\S]*?)\\]\\n*', 'g');
    };
    content = (content || '').replace(/&(?!#?[a-zA-Z0-9]+;)/g, '&amp;')
        .replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/'/g, '&#39;').replace(/"/g, '&quot;') // XSS
        .replace(/@(\S+)(\s+?|$)/g, '@<a href="javascript:;">$1</a>$2') // 转义@

        .replace(/face\[([^\s\[\]]+?)\]/g, function (face) {  // 转义表情
            var alt = face.replace(/^face/g, '');
            return '<img alt="' + alt + '" title="' + alt + '" src="' + faces[alt] + '">';
        })
        .replace(/img\[([^\s]+?)\]/g, function (img) {  // 转义图片
            return '<img class="layui-AiService-photos" src="' + img.replace(/(^img\[)|(\]$)/g, '') + '" style="max-width: 100%;width: 100%;height: 150px">';
        })
        .replace(/file\([\s\S]+?\)\[[\s\S]*?\]/g, function (str) { // 转义文件
            var href = (str.match(/file\(([\s\S]+?)\)\[/) || [])[1];
            var text = (str.match(/\)\[([\s\S]*?)\]/) || [])[1];
            if (!href) return str;
            return '<a class="layui-AiService-file" href="' + href + '" download target="_blank"><i class="layui-icon">&#xe61e;</i><cite>' + (text || href) + '</cite></a>';
        })
        .replace(/audio\[([^\s]+?)\]/g, function(audio){  //转义音频
            return '<audio src="' + audio.replace(/(^audio\[)|(\]$)/g, '') + '" controls="controls" style="width: 200px;height: 20px"></audio>';
        })
        .replace(/a\([\s\S]+?\)\[[\s\S]*?\]/g, function (str) { // 转义链接
            var href = (str.match(/a\(([\s\S]+?)\)\[/) || [])[1];
            var text = (str.match(/\)\[([\s\S]*?)\]/) || [])[1];
            if (!href) return str;
            return '<a href="' + href + '" target="_blank" style="color:#1E9FFF">' + (text || href) + '</a>';
        }).replace(html(), '\<$1 $2\>').replace(html('/'), '\</$1\>') // 转移HTML代码
        .replace(/\n/g, '<br>');// 转义换行

    return content;
}
